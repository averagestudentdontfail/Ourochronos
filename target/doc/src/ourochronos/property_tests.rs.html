15    proptest! {
16        /// Memory hash is consistent: same writes produce same hash.
17        #[test]
18        fn prop_memory_hash_deterministic(
19            addr1 in 0u16..1000,
20            val1 in any::<u64>(),
21            addr2 in 0u16..1000,
22            val2 in any::<u64>(),
23        ) {
24            let mut mem1 = Memory::new();
25            let mut mem2 = Memory::new();
26            
27            mem1.write(addr1, Value::new(val1));
28            mem1.write(addr2, Value::new(val2));
29            
30            mem2.write(addr1, Value::new(val1));
31            mem2.write(addr2, Value::new(val2));
32            
33            prop_assert_eq!(mem1.state_hash(), mem2.state_hash());
34        }
35        
36        /// Incremental hash matches full recompute.
37        #[test]
38        fn prop_incremental_hash_correct(
39            writes in prop::collection::vec((0u16..500, any::<u64>()), 1..20),
40        ) {
41            let mut mem = Memory::new();
42            
43            for (addr, val) in writes {
44                mem.write(addr, Value::new(val));
45            }
46            
47            prop_assert_eq!(mem.state_hash(), mem.recompute_hash());
48        }
49        
50        /// Memory ordering is antisymmetric.
51        #[test]
52        fn prop_memory_ordering_antisymmetric(
53            addr in 0u16..100,
54            val1 in any::<u64>(),
55            val2 in any::<u64>(),
56        ) {
57            let mut mem1 = Memory::new();
58            let mut mem2 = Memory::new();
59            
60            mem1.write(addr, Value::new(val1));
61            mem2.write(addr, Value::new(val2));
62            
63            use std::cmp::Ordering;
64            let cmp1 = mem1.cmp(&mem2);
65            let cmp2 = mem2.cmp(&mem1);
66            
67            match cmp1 {
68                Ordering::Less => prop_assert_eq!(cmp2, Ordering::Greater),
69                Ordering::Greater => prop_assert_eq!(cmp2, Ordering::Less),
70                Ordering::Equal => prop_assert_eq!(cmp2, Ordering::Equal),
71            }
72        }
73    }

121    proptest! {
122        /// Selection is idempotent: selecting from same candidates gives same result.
123        #[test]
124        fn prop_selection_idempotent(
125            vals in prop::collection::vec(0u64..1000, 2..10),
126        ) {
127            use crate::action::{ActionPrinciple, ActionConfig, FixedPointSelector};
128            
129            let principle = ActionPrinciple::new(ActionConfig::default());
130            let seed = Memory::new();
131            
132            let mut results = Vec::new();
133            
134            for _ in 0..3 {
135                let mut selector = FixedPointSelector::new(principle.clone());
136                
137                for (i, &val) in vals.iter().enumerate() {
138                    let mut mem = Memory::new();
139                    mem.write(0, Value::new(val));
140                    mem.write(1, Value::new(i as u64));
141                    selector.add_candidate(mem, 1, vec![], seed.clone());
142                }
143                
144                if let Some(best) = selector.select_best() {
145                    results.push(best.memory.read(0).val);
146                }
147            }
148            
149            // All selections should be identical
150            if let Some(&first) = results.first() {
151                for &val in &results {
152                    prop_assert_eq!(val, first);
153                }
154            }
155        }
156    }

162    proptest! {
163        /// Pure programs (no ORACLE) always converge in 1 epoch.
164        #[test]
165        fn prop_pure_program_single_epoch(
166            a in 1u64..1000,
167            b in 1u64..1000,
168        ) {
169            let source = format!("{} {} ADD OUTPUT", a, b);
170            let tokens = tokenize(&source);
171            let mut parser = Parser::new(&tokens);
172            
173            if let Ok(program) = parser.parse_program() {
174                let config = timeloop::TimeLoopConfig::default();
175                let mut driver = TimeLoop::new(config);
176                let result = driver.run(&program);
177                
178                match result {
179                    ConvergenceStatus::Consistent { epochs, output, .. } => {
180                        prop_assert_eq!(epochs, 1);
181                        prop_assert!(!output.is_empty());
182                        prop_assert_eq!(output[0].val, a.wrapping_add(b));
183                    }
184                    _ => prop_assert!(false, "Expected consistent"),
185                }
186            }
187        }
188    }
