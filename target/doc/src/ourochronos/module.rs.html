1//! Module system for OUROCHRONOS.
2//!
3//! Provides support for organizing code into reusable modules with imports/exports.
4//!
5//! # Syntax
6//!
7//! ```ourochronos
8//! # primality.ouro
9//! MODULE primality;
10//!
11//! EXPORT PROCEDURE is_factor { ... }
12//!
13//! # main.ouro
14//! IMPORT primality;
15//!
16//! 15 3 primality::is_factor IF { ... }
17//! ```

19use std::collections::HashMap;
20use std::path::PathBuf;

23use crate::ast::Procedure;

25/// A module definition.
26#[derive(Debug, Clone)]
27pub struct Module {
28    /// Module name.
29    pub name: String,
30    /// Exported procedures.
31    pub exports: Vec<String>,
32    /// All procedures defined in the module.
33    pub procedures: Vec<Procedure>,
34    /// Source file path.
35    pub source_path: PathBuf,
36}
37
38impl Module {
39    /// Create a new module with the given name.
40    pub fn new(name: String, source_path: PathBuf) -> Self {
41        Self {
42            name,
43            exports: Vec::new(),
44            procedures: Vec::new(),
45            source_path,
46        }
47    }
48    
49    /// Check if a procedure is exported.
50    pub fn is_exported(&self, proc_name: &str) -> bool {
51        self.exports.contains(&proc_name.to_string()) || self.exports.is_empty()
52    }
53    
54    /// Get an exported procedure by name.
55    pub fn get_procedure(&self, name: &str) -> Option<&Procedure> {
56        self.procedures.iter().find(|p| p.name == name)
57    }
58}

60/// Module registry for tracking loaded modules.
61#[derive(Debug, Default)]
62pub struct ModuleRegistry {
63    /// Map from module name to module.
64    modules: HashMap<String, Module>,
65    /// Search paths for module resolution.
66    search_paths: Vec<PathBuf>,
67}
68
69impl ModuleRegistry {
70    /// Create a new empty registry.
71    pub fn new() -> Self {
72        Self::default()
73    }
74    
75    /// Add a search path for module resolution.
76    pub fn add_search_path(&mut self, path: PathBuf) {
77        self.search_paths.push(path);
78    }
79    
80    /// Register a module.
81    pub fn register(&mut self, module: Module) {
82        self.modules.insert(module.name.clone(), module);
83    }
84    
85    /// Get a module by name.
86    pub fn get(&self, name: &str) -> Option<&Module> {
87        self.modules.get(name)
88    }
89    
90    /// Resolve a qualified name (module::procedure) to a procedure.
91    pub fn resolve_qualified(&self, qualified_name: &str) -> Option<&Procedure> {
92        let parts: Vec<&str> = qualified_name.split("::").collect();
93        if parts.len() != 2 {
94            return None;
95        }
96        
97        let module_name = parts[0];
98        let proc_name = parts[1];
99        
100        self.get(module_name)
101            .filter(|m| m.is_exported(proc_name))
102            .and_then(|m| m.get_procedure(proc_name))
103    }
104    
105    /// Load a module from file.
106    pub fn load_module(&mut self, name: &str) -> Result<(), String> {
107        // Search for module file
108        let filename = format!("{}.ouro", name);
109        
110        let mut found_path: Option<PathBuf> = None;
111        for search_path in &self.search_paths {
112            let candidate = search_path.join(&filename);
113            if candidate.exists() {
114                found_path = Some(candidate);
115                break;
116            }
117        }
118        
119        let path = found_path.ok_or_else(|| format!("Module not found: {}", name))?;
120        
121        // Read and parse
122        let source = fs::read_to_string(&path)
123            .map_err(|e| format!("Cannot read module {}: {}", name, e))?;
124        
125        // Parse the module (simplified - would need to extract MODULE/EXPORT declarations)
126        let program = crate::parser::parse(&source)?;
127        
128        let mut module = Module::new(name.to_string(), path);
129        module.procedures = program.procedures;
130        
131        // Mark all procedures as exported (simplified)
132        module.exports = module.procedures.iter().map(|p| p.name.clone()).collect();
133        
134        self.register(module);
135        Ok(())
136    }
137    
138    /// List all registered modules.
139    pub fn list_modules(&self) -> Vec<&str> {
140        self.modules.keys().map(|s| s.as_str()).collect()
141    }
142}

144/// Parse MODULE declaration from source.
145pub fn parse_module_declaration(source: &str) -> Option<String> {
146    for line in source.lines() {
147        let line = line.trim();
148        if line.starts_with("MODULE") {
149            // Extract module name
150            let rest = line.strip_prefix("MODULE")?.trim();
151            let name = rest.trim_end_matches(';').trim();
152            if !name.is_empty() {
153                return Some(name.to_string());
154            }
155        }
156        // Stop at first non-comment, non-declaration line
157        if !line.is_empty() && !line.starts_with('#') && !line.starts_with("MODULE") {
158            break;
159        }
160    }
161    None
162}
163
164/// Parse EXPORT declarations from source.
165pub fn parse_exports(source: &str) -> Vec<String> {
166    let mut exports = Vec::new();
167    
168    for line in source.lines() {
169        let line = line.trim();
170        if line.starts_with("EXPORT") {
171            // EXPORT PROCEDURE name or EXPORT name
172            let rest = line.strip_prefix("EXPORT").unwrap().trim();
173            let rest = rest.strip_prefix("PROCEDURE").unwrap_or(rest).trim();
174            let name = rest.trim_end_matches(';').trim();
175            if let Some(name) = name.split_whitespace().next() {
176                exports.push(name.to_lowercase());
177            }
178        }
179    }
180    
181    exports
182}

184#[cfg(test)]

188    #[test]
189    fn test_parse_module_declaration() {
190        assert_eq!(
191            parse_module_declaration("MODULE math;"),
192            Some("math".to_string())
193        );
194        assert_eq!(
195            parse_module_declaration("# comment\nMODULE crypto;"),
196            Some("crypto".to_string())
197        );
198        assert_eq!(
199            parse_module_declaration("10 20 ADD"),
200            None
201        );
202    }
203    
204    #[test]
205    fn test_parse_exports() {
206        let source = "
207            EXPORT PROCEDURE gcd;
208            EXPORT is_prime;
209            PROCEDURE helper { }
210        ";
211        let exports = parse_exports(source);
212        assert!(exports.contains(&"gcd".to_string()));
213        assert!(exports.contains(&"is_prime".to_string()));
214        assert!(!exports.contains(&"helper".to_string()));
215    }
