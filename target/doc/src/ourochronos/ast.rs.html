1//! Abstract Syntax Tree for OUROCHRONOS.
2//!
3//! OUROCHRONOS uses a stack-based execution model with structured control flow.
4//! Programs consist of statements that manipulate a value stack, present memory,
5//! and can read from anamnesis (the temporal oracle).
6
7use crate::core_types::Value;
8
9/// All operations available in OUROCHRONOS.
10#[derive(Debug, Clone, Copy, PartialEq, Eq)]
11pub enum OpCode {
12    // ═══════════════════════════════════════════════════════════════════
13    // Stack Manipulation
14    // ═══════════════════════════════════════════════════════════════════
15    
16    /// No operation.
17    Nop,
18    
19    /// Halt execution of the current epoch.
20    /// Stack: ( -- )
21    Halt,
22    
23    /// Pop and discard the top of stack.
24    /// Stack: ( a -- )
25    Pop,
26    
27    /// Duplicate the top of stack.
28    /// Stack: ( a -- a a )
29    Dup,
30    
31    /// Swap the top two elements.
32    /// Stack: ( a b -- b a )
33    Swap,
34    
35    /// Copy the second element to the top.
36    /// Stack: ( a b -- a b a )
37    Over,
38    
39    /// Rotate the top three elements.
40    /// Stack: ( a b c -- b c a )
41    Rot,
42    
43    /// Push the current stack depth.
44    /// Stack: ( -- n )
45    Depth,
46
47    /// Pick the nth element from deep in the stack and copy it to top.
48    /// 0 PICK is equivalent to DUP. 1 PICK is equivalent to OVER.
49    /// Stack: ( n -- v )
50    Pick,
51    
52    // ═══════════════════════════════════════════════════════════════════
53    // Arithmetic (modular, wrapping at 2^64)
54    // ═══════════════════════════════════════════════════════════════════
55    
56    /// Addition.
57    /// Stack: ( a b -- a+b )
58    Add,
59    
60    /// Subtraction.
61    /// Stack: ( a b -- a-b )
62    Sub,
63    
64    /// Multiplication.
65    /// Stack: ( a b -- a*b )
66    Mul,
67    
68    /// Division (returns 0 if divisor is 0).
69    /// Stack: ( a b -- a/b )
70    Div,
71    
72    /// Modulo (returns 0 if divisor is 0).
73    /// Stack: ( a b -- a%b )
74    Mod,
75    
76    /// Negation (two's complement).
77    /// Stack: ( a -- -a )
78    Neg,
79    
80    // ═══════════════════════════════════════════════════════════════════
81    // Bitwise Logic
82    // ═══════════════════════════════════════════════════════════════════
83    
84    /// Bitwise NOT.
85    /// Stack: ( a -- ~a )
86    Not,
87    
88    /// Bitwise AND.
89    /// Stack: ( a b -- a&b )
90    And,
91    
92    /// Bitwise OR.
93    /// Stack: ( a b -- a|b )
94    Or,
95    
96    /// Bitwise XOR.
97    /// Stack: ( a b -- a^b )
98    Xor,
99    
100    /// Left shift.
101    /// Stack: ( a n -- a<<n )
102    Shl,
103    
104    /// Right shift (logical).
105    /// Stack: ( a n -- a>>n )
106    Shr,
107    
108    // ═══════════════════════════════════════════════════════════════════
109    // Comparison (result: 1 if true, 0 if false)
110    // ═══════════════════════════════════════════════════════════════════
111    
112    /// Equal.
113    /// Stack: ( a b -- a==b )
114    Eq,
115    
116    /// Not equal.
117    /// Stack: ( a b -- a!=b )
118    Neq,
119    
120    /// Less than.
121    /// Stack: ( a b -- a<b )
122    Lt,
123    
124    /// Greater than.
125    /// Stack: ( a b -- a>b )
126    Gt,
127    
128    /// Less than or equal.
129    /// Stack: ( a b -- a<=b )
130    Lte,
131    
132    /// Greater than or equal.
133    /// Stack: ( a b -- a>=b )
134    Gte,
135    
136    // ═══════════════════════════════════════════════════════════════════
137    // Temporal Operations (the core of OUROCHRONOS)
138    // ═══════════════════════════════════════════════════════════════════
139    
140    /// ORACLE: Read from anamnesis (the future).
141    /// Pops an address, pushes the value from Anamnesis[address].
142    /// This is the mechanism for receiving information from the future.
143    /// Stack: ( addr -- A[addr] )
144    Oracle,
145    
146    /// PROPHECY: Write to present (fulfilling the future).
147    /// Pops value and address, writes value to Present[address].
148    /// This is the mechanism for sending information to the past.
149    /// Stack: ( value addr -- )
150    Prophecy,
151    
152    /// PRESENT_READ: Read from present memory (current epoch).
153    /// Stack: ( addr -- P[addr] )
154    PresentRead,
155    
156    /// PARADOX: Signal explicit inconsistency.
157    /// Terminates the current epoch as paradoxical.
158    /// In fixed-point search, this path is rejected.
159    /// Stack: ( -- )
160    Paradox,
161    
162    // ═══════════════════════════════════════════════════════════════════
163    // Array/Memory Operations
164    // ═══════════════════════════════════════════════════════════════════
165    
166    /// Pack n values into contiguous memory starting at base address.
167    /// Stack: ( v1 v2 ... vn base n -- )
168    Pack,
169    
170    /// Unpack n values from contiguous memory at base address.
171    /// Stack: ( base n -- v1 v2 ... vn )
172    Unpack,
173    
174    /// Read from memory at computed index: base + index.
175    /// Stack: ( base index -- P[base+index] )
176    Index,
177    
178    /// Store to memory at computed index: base + index.
179    /// Stack: ( value base index -- )
180    Store,
181    
182    // ═══════════════════════════════════════════════════════════════════
183    // I/O Operations
184    // ═══════════════════════════════════════════════════════════════════
185    
186    /// Read a value from input.
187    /// Stack: ( -- value )
188    Input,
189    
190    /// Output a value.
191    /// Stack: ( value -- )
192    Output,
193}

195impl OpCode {
196    /// Get the name of this opcode as it appears in source code.
197    pub fn name(&self) -> &'static str {
198        match self {
199            OpCode::Nop => "NOP",
200            OpCode::Halt => "HALT",
201            OpCode::Pop => "POP",
202            OpCode::Dup => "DUP",
203            OpCode::Swap => "SWAP",
204            OpCode::Over => "OVER",
205            OpCode::Rot => "ROT",
206            OpCode::Depth => "DEPTH",
207            OpCode::Pick => "PICK",
208            OpCode::Add => "ADD",
209            OpCode::Sub => "SUB",
210            OpCode::Mul => "MUL",
211            OpCode::Div => "DIV",
212            OpCode::Mod => "MOD",
213            OpCode::Neg => "NEG",
214            OpCode::Not => "NOT",
215            OpCode::And => "AND",
216            OpCode::Or => "OR",
217            OpCode::Xor => "XOR",
218            OpCode::Shl => "SHL",
219            OpCode::Shr => "SHR",
220            OpCode::Eq => "EQ",
221            OpCode::Neq => "NEQ",
222            OpCode::Lt => "LT",
223            OpCode::Gt => "GT",
224            OpCode::Lte => "LTE",
225            OpCode::Gte => "GTE",
226            OpCode::Oracle => "ORACLE",
227            OpCode::Prophecy => "PROPHECY",
228            OpCode::PresentRead => "PRESENT",
229            OpCode::Paradox => "PARADOX",
230            OpCode::Pack => "PACK",
231            OpCode::Unpack => "UNPACK",
232            OpCode::Index => "INDEX",
233            OpCode::Store => "STORE",
234            OpCode::Input => "INPUT",
235            OpCode::Output => "OUTPUT",
236        }
237    }
238    
239    /// Get the stack effect: (inputs, outputs).
240    pub fn stack_effect(&self) -> (usize, usize) {
241        match self {
242            OpCode::Nop | OpCode::Halt | OpCode::Paradox => (0, 0),
243            OpCode::Pop | OpCode::Output => (1, 0),
244            OpCode::Dup => (1, 2),
245            OpCode::Swap => (2, 2),
246            OpCode::Over => (2, 3),
247
248            OpCode::Rot => (3, 3),
249            OpCode::Depth | OpCode::Input => (0, 1),
250            OpCode::Pick => (1, 1),
251            OpCode::Neg | OpCode::Not => (1, 1),
252            OpCode::Add | OpCode::Sub | OpCode::Mul | OpCode::Div | OpCode::Mod |
253            OpCode::And | OpCode::Or | OpCode::Xor | OpCode::Shl | OpCode::Shr |
254            OpCode::Eq | OpCode::Neq | OpCode::Lt | OpCode::Gt | OpCode::Lte | OpCode::Gte => (2, 1),
255            OpCode::Oracle | OpCode::PresentRead => (1, 1),
256            OpCode::Prophecy => (2, 0),
257            // Array opcodes have variable effects, these are minimums
258            OpCode::Pack => (2, 0),    // base, n, (plus n values consumed)
259            OpCode::Unpack => (2, 0),  // base, n (produces n values)
260            OpCode::Index => (2, 1),   // base, index -> value
261            OpCode::Store => (3, 0),   // value, base, index
262        }
263    }
264}

266/// A statement in the OUROCHRONOS AST.
267#[derive(Debug, Clone, PartialEq)]
268pub enum Stmt {
269    /// Execute a single opcode.
270    Op(OpCode),
271    
272    /// Push a constant value onto the stack.
273    Push(Value),
274    
275    /// Structured IF statement.
276    /// Pops the condition from stack; if non-zero, executes then_branch.
277    If {
278        then_branch: Vec<Stmt>,
279        else_branch: Option<Vec<Stmt>>,
280    },
281    
282    /// Structured WHILE loop.
283    /// Evaluates cond block, pops result; if non-zero, executes body and repeats.
284    While {
285        cond: Vec<Stmt>,
286        body: Vec<Stmt>,
287    },
288    
289    /// A block of statements (scoped grouping).
290    Block(Vec<Stmt>),
291    
292    /// Call a procedure by name.
293    /// Stack effect depends on the procedure's parameter and return count.
294    Call {
295        name: String,
296    },
297    
298    /// Pattern match on top-of-stack value.
299    /// Pops value and selects matching case branch or default.
300    Match {
301        /// List of (pattern, body) pairs.
302        cases: Vec<(u64, Vec<Stmt>)>,
303        /// Default case if no pattern matches.
304        default: Option<Vec<Stmt>>,
305    },
306    
307    /// Temporal scope block.
308    /// TEMPORAL <base> <size> { body }
309    /// 
310    /// Creates an isolated memory region for temporal operations.
311    /// ORACLE/PROPHECY within the block are relative to base address.
312    /// Changes within the scope are propagated to parent on successful exit.
313    /// If the block induces a paradox, changes are discarded.
314    TemporalScope {
315        /// Base address for the isolated region.
316        base: u64,
317        /// Size of the isolated region.
318        size: u64,
319        /// Body of the temporal scope.
320        body: Vec<Stmt>,
321    },
322}

324/// Side effect behavior of a procedure.
325#[derive(Debug, Clone, PartialEq, Eq, Hash)]
326pub enum Effect {
327    /// Procedure is pure (no side effects, deterministic given inputs).
328    /// Implies no ORACLE reads and no PROPHECY writes.
329    Pure,
330    /// Procedure reads from the specific oracle address.
331    Reads(u64),
332    /// Procedure writes to the specific prophecy address.
333    Writes(u64),
334}
335
336/// A procedure definition.
337#[derive(Debug, Clone, PartialEq)]
338pub struct Procedure {
339    /// Procedure name.
340    pub name: String,
341    /// Parameter names (for documentation; all params come from stack).
342    pub params: Vec<String>,
343    /// Number of return values pushed to stack.
344    pub returns: usize,
345    /// Effect annotations declared by the user.
346    pub effects: Vec<Effect>,
347    /// Body of the procedure.
348    pub body: Vec<Stmt>,
349}
350
351/// A complete OUROCHRONOS program.
352#[derive(Debug, Clone)]
353pub struct Program {
354    /// Procedure definitions.
355    pub procedures: Vec<Procedure>,
356    /// Main program body.
357    pub body: Vec<Stmt>,
358}
359
360impl Program {
361    /// Create an empty program.
362    pub fn new() -> Self {
363        Program { 
364            procedures: Vec::new(),
365            body: Vec::new(),
366        }
367    }
368    
369    /// Check if the program is trivially consistent (no oracle operations).
370    pub fn is_trivially_consistent(&self) -> bool {
371        !self.contains_oracle(&self.body)
372    }
373    
374    fn contains_oracle(&self, stmts: &[Stmt]) -> bool {
375        for stmt in stmts {
376            match stmt {
377                Stmt::Op(OpCode::Oracle) => return true,
378                Stmt::If { then_branch, else_branch } => {
379                    if self.contains_oracle(then_branch) {
380                        return true;
381                    }
382                    if let Some(else_stmts) = else_branch {
383                        if self.contains_oracle(else_stmts) {
384                            return true;
385                        }
386                    }
387                }
388                Stmt::While { cond, body } => {
389                    if self.contains_oracle(cond) || self.contains_oracle(body) {
390                        return true;
391                    }
392                }
393                Stmt::Block(inner) => {
394                    if self.contains_oracle(inner) {
395                        return true;
396                    }
397                }
398                Stmt::TemporalScope { body, .. } => {
399                    if self.contains_oracle(body) {
400                        return true;
401                    }
402                }
403                _ => {}
404            }
405        }
406        false
407    }
408    
409    /// Inline all procedure calls, replacing Stmt::Call with procedure bodies.
410    /// Returns a new Program with all calls inlined.
411    pub fn inline_procedures(&self) -> Self {
412        let proc_map: std::collections::HashMap<String, &Procedure> = 
413            self.procedures.iter().map(|p| (p.name.clone(), p)).collect();
414        
415        Program {
416            procedures: Vec::new(), // Procedures are now inlined
417            body: self.inline_stmts(&self.body, &proc_map),
418        }
419    }
420    
421    fn inline_stmts(&self, stmts: &[Stmt], procs: &std::collections::HashMap<String, &Procedure>) -> Vec<Stmt> {
422        stmts.iter().map(|stmt| self.inline_stmt(stmt, procs)).collect()
423    }
424    
425    fn inline_stmt(&self, stmt: &Stmt, procs: &std::collections::HashMap<String, &Procedure>) -> Stmt {
426        match stmt {
427            Stmt::Call { name } => {
428                if let Some(proc) = procs.get(name) {
429                    // Inline the procedure body
430                    Stmt::Block(self.inline_stmts(&proc.body, procs))
431                } else {
432                    // Procedure not found - keep as is (will error at runtime)
433                    stmt.clone()
434                }
435            }
436            Stmt::If { then_branch, else_branch } => Stmt::If {
437                then_branch: self.inline_stmts(then_branch, procs),
438                else_branch: else_branch.as_ref().map(|e| self.inline_stmts(e, procs)),
439            },
440            Stmt::While { cond, body } => Stmt::While {
441                cond: self.inline_stmts(cond, procs),
442                body: self.inline_stmts(body, procs),
443            },
444            Stmt::Block(inner) => Stmt::Block(self.inline_stmts(inner, procs)),
445            Stmt::TemporalScope { base, size, body } => Stmt::TemporalScope {
446                base: *base,
447                size: *size,
448                body: self.inline_stmts(body, procs),
449            },
450            other => other.clone(),
451        }
452    }
453}
454
455impl Default for Program {
456    fn default() -> Self {
457        Self::new()
458    }
