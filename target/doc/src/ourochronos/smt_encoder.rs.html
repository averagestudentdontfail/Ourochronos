1//! SMT-LIB2 encoder for OUROCHRONOS programs.
2//!
3//! Compiles the fixed-point constraint A = F(A) to SMT-LIB2 format,
4//! allowing industrial solvers (Z3, CVC5) to find fixed points or
5//! prove paradoxes (UNSAT).
6
7use crate::ast::{Program, Stmt, OpCode};
8use std::fmt::Write;
9
10/// SMT-LIB2 encoder for OUROCHRONOS.
11pub struct SmtEncoder {
12    /// Generated SMT code.
13    output: String,
14    /// Counter for fresh variable names.
15    var_counter: usize,
16    /// Maximum loop unrolling depth.
17    max_unroll: usize,
18}
19
20impl SmtEncoder {
21    /// Create a new encoder.
22    pub fn new() -> Self {
23        Self {
24            output: String::new(),
25            var_counter: 0,
26            max_unroll: 10,
27        }
28    }
29    
30    /// Create an encoder with custom unroll limit.
31    pub fn with_unroll_limit(max_unroll: usize) -> Self {
32        Self {
33            output: String::new(),
34            var_counter: 0,
35            max_unroll,
36        }
37    }
38    
39    /// Encode a program to SMT-LIB2.
40    pub fn encode(&mut self, program: &Program) -> String {
41        self.output.clear();
42        self.var_counter = 0;
43        
44        // Header
45        writeln!(self.output, "; OUROCHRONOS SMT-LIB2 Encoding").unwrap();
46        writeln!(self.output, "; Fixed-point constraint: A = F(A)").unwrap();
47        writeln!(self.output).unwrap();
48        
49        // Logic: Quantifier-Free Arrays and Bit-Vectors
50        writeln!(self.output, "(set-logic QF_ABV)").unwrap();
51        writeln!(self.output, "(set-option :produce-models true)").unwrap();
52        writeln!(self.output).unwrap();
53        
54        // Declare Anamnesis (the unknown we're solving for)
55        writeln!(self.output, "; Anamnesis: the memory state from the 'future'").unwrap();
56        writeln!(self.output, "(declare-const anamnesis (Array (_ BitVec 16) (_ BitVec 64)))").unwrap();
57        writeln!(self.output).unwrap();
58        
59        // Initialize Present to all zeros
60        writeln!(self.output, "; Present: starts as all zeros").unwrap();
61        writeln!(self.output, "(declare-const present_init (Array (_ BitVec 16) (_ BitVec 64)))").unwrap();
62        writeln!(self.output, "(assert (= present_init ((as const (Array (_ BitVec 16) (_ BitVec 64))) (_ bv0 64))))").unwrap();
63        writeln!(self.output).unwrap();
64        
65        // Symbolic execution
66        writeln!(self.output, "; Symbolic execution of program").unwrap();
67        let mut stack: Vec<String> = Vec::new();
68        let mut present = "present_init".to_string();
69        
70        self.encode_block(&program.body, &mut stack, &mut present, 0);
71        
72        writeln!(self.output).unwrap();
73        
74        // Fixed-point constraint
75        writeln!(self.output, "; Fixed-point constraint: Present = Anamnesis").unwrap();
76        writeln!(self.output, "(assert (= {} anamnesis))", present).unwrap();
77        writeln!(self.output).unwrap();
78        
79        // Check and get model
80        writeln!(self.output, "(check-sat)").unwrap();
81        writeln!(self.output, "(get-model)").unwrap();
82        
83        self.output.clone()
84    }
85    
86    /// Generate a fresh variable name.
87    fn fresh_var(&mut self, prefix: &str) -> String {
88        let name = format!("{}_{}", prefix, self.var_counter);
89        self.var_counter += 1;
90        name
91    }
92    
93    /// Encode a block of statements.
94    fn encode_block(&mut self, stmts: &[Stmt], 
95                    stack: &mut Vec<String>, 
96                    present: &mut String,
97                    depth: usize) {
98        for stmt in stmts {
99            self.encode_stmt(stmt, stack, present, depth);
100        }
101    }
102    
103    /// Encode a single statement.
104    fn encode_stmt(&mut self, stmt: &Stmt, 
105                   stack: &mut Vec<String>, 
106                   present: &mut String,
107                   depth: usize) {
108        match stmt {
109            Stmt::Push(v) => {
110                stack.push(format!("(_ bv{} 64)", v.val));
111            }
112            
113            Stmt::Op(op) => {
114                self.encode_op(*op, stack, present);
115            }
116            
117            Stmt::Block(stmts) => {
118                self.encode_block(stmts, stack, present, depth);
119            }
120            
121            Stmt::If { then_branch, else_branch } => {
122                self.encode_if(then_branch, else_branch.as_deref(), stack, present, depth);
123            }
124            
125            Stmt::While { cond, body } => {
126                self.encode_while(cond, body, stack, present, depth);
127            }
128            
129            Stmt::Call { name } => {
130                panic!("SMT encoding does not support procedure calls - inline procedures first: {}", name);
131            }
132            
133            Stmt::Match { cases: _, default: _ } => {
134                // Pattern matching requires complex ITE chains
135                // For now, skip - would need full stack simulation
136            }
137            
138            Stmt::TemporalScope { body, .. } => {
139                // Temporal scoping: encode body, but note isolation semantics
140                // are not fully expressible in SMT without memory regions
141                self.encode_block(body, stack, present, depth);
142            }
143        }
144    }

146    /// Encode an opcode.
147    fn encode_op(&mut self, op: OpCode, stack: &mut Vec<String>, present: &mut String) {
148        match op {
149            OpCode::Nop | OpCode::Halt | OpCode::Paradox => {}
150            
151            OpCode::Pop => { stack.pop(); }
152            
153            OpCode::Dup => {
154                if let Some(top) = stack.last().cloned() {
155                    stack.push(top);
156                }
157            }
158            
159            OpCode::Swap => {
160                if stack.len() >= 2 {
161                    let len = stack.len();
162                    stack.swap(len - 1, len - 2);
163                }
164            }
165            
166            OpCode::Over => {
167                if stack.len() >= 2 {
168                    let val = stack[stack.len() - 2].clone();
169                    stack.push(val);
170                }
171            }
172            
173            OpCode::Rot => {
174                if stack.len() >= 3 {
175                    let len = stack.len();
176                    let c = stack.remove(len - 3);
177                    stack.push(c);
178                }
179            }
180            
181            OpCode::Depth => {
182                stack.push(format!("(_ bv{} 64)", stack.len()));
183            }
184            OpCode::Pick => {
185                 panic!("SMT encoding for PICK not implemented");
186            }
187            OpCode::Add => {
188                if stack.len() >= 2 {
189                    let b = stack.pop().unwrap();
190                    let a = stack.pop().unwrap();
191                    stack.push(format!("(bvadd {} {})", a, b));
192                }
193            }
194            
195            OpCode::Sub => {
196                if stack.len() >= 2 {
197                    let b = stack.pop().unwrap();
198                    let a = stack.pop().unwrap();
199                    stack.push(format!("(bvsub {} {})", a, b));
200                }
201            }
202            
203            OpCode::Mul => {
204                if stack.len() >= 2 {
205                    let b = stack.pop().unwrap();
206                    let a = stack.pop().unwrap();
207                    stack.push(format!("(bvmul {} {})", a, b));
208                }
209            }
210            
211            OpCode::Div => {
212                if stack.len() >= 2 {
213                    let b = stack.pop().unwrap();
214                    let a = stack.pop().unwrap();
215                    // Handle div by zero: (ite (= b 0) 0 (bvudiv a b))
216                    stack.push(format!(
217                        "(ite (= {} (_ bv0 64)) (_ bv0 64) (bvudiv {} {}))",
218                        b, a, b
219                    ));
220                }
221            }
222            
223            OpCode::Mod => {
224                if stack.len() >= 2 {
225                    let b = stack.pop().unwrap();
226                    let a = stack.pop().unwrap();
227                    stack.push(format!(
228                        "(ite (= {} (_ bv0 64)) (_ bv0 64) (bvurem {} {}))",
229                        b, a, b
230                    ));
231                }
232            }
233            
234            OpCode::Neg => {
235                if let Some(a) = stack.pop() {
236                    stack.push(format!("(bvneg {})", a));
237                }
238            }
239            
240            OpCode::Not => {
241                if let Some(a) = stack.pop() {
242                    stack.push(format!("(bvnot {})", a));
243                }
244            }
245            
246            OpCode::And => {
247                if stack.len() >= 2 {
248                    let b = stack.pop().unwrap();
249                    let a = stack.pop().unwrap();
250                    stack.push(format!("(bvand {} {})", a, b));
251                }
252            }
253            
254            OpCode::Or => {
255                if stack.len() >= 2 {
256                    let b = stack.pop().unwrap();
257                    let a = stack.pop().unwrap();
258                    stack.push(format!("(bvor {} {})", a, b));
259                }
260            }
261            
262            OpCode::Xor => {
263                if stack.len() >= 2 {
264                    let b = stack.pop().unwrap();
265                    let a = stack.pop().unwrap();
266                    stack.push(format!("(bvxor {} {})", a, b));
267                }
268            }
269            
270            OpCode::Shl => {
271                if stack.len() >= 2 {
272                    let n = stack.pop().unwrap();
273                    let a = stack.pop().unwrap();
274                    stack.push(format!("(bvshl {} {})", a, n));
275                }
276            }
277            
278            OpCode::Shr => {
279                if stack.len() >= 2 {
280                    let n = stack.pop().unwrap();
281                    let a = stack.pop().unwrap();
282                    stack.push(format!("(bvlshr {} {})", a, n));
283                }
284            }
285            
286            OpCode::Eq => {
287                if stack.len() >= 2 {
288                    let b = stack.pop().unwrap();
289                    let a = stack.pop().unwrap();
290                    stack.push(format!("(ite (= {} {}) (_ bv1 64) (_ bv0 64))", a, b));
291                }
292            }
293            
294            OpCode::Neq => {
295                if stack.len() >= 2 {
296                    let b = stack.pop().unwrap();
297                    let a = stack.pop().unwrap();
298                    stack.push(format!("(ite (not (= {} {})) (_ bv1 64) (_ bv0 64))", a, b));
299                }
300            }
301            
302            OpCode::Lt => {
303                if stack.len() >= 2 {
304                    let b = stack.pop().unwrap();
305                    let a = stack.pop().unwrap();
306                    stack.push(format!("(ite (bvult {} {}) (_ bv1 64) (_ bv0 64))", a, b));
307                }
308            }
309            
310            OpCode::Gt => {
311                if stack.len() >= 2 {
312                    let b = stack.pop().unwrap();
313                    let a = stack.pop().unwrap();
314                    stack.push(format!("(ite (bvugt {} {}) (_ bv1 64) (_ bv0 64))", a, b));
315                }
316            }
317            
318            OpCode::Lte => {
319                if stack.len() >= 2 {
320                    let b = stack.pop().unwrap();
321                    let a = stack.pop().unwrap();
322                    stack.push(format!("(ite (bvule {} {}) (_ bv1 64) (_ bv0 64))", a, b));
323                }
324            }
325            
326            OpCode::Gte => {
327                if stack.len() >= 2 {
328                    let b = stack.pop().unwrap();
329                    let a = stack.pop().unwrap();
330                    stack.push(format!("(ite (bvuge {} {}) (_ bv1 64) (_ bv0 64))", a, b));
331                }
332            }
333            
334            OpCode::Oracle => {
335                if let Some(addr) = stack.pop() {
336                    let addr16 = format!("((_ extract 15 0) {})", addr);
337                    stack.push(format!("(select anamnesis {})", addr16));
338                }
339            }
340            
341            OpCode::Prophecy => {
342                if stack.len() >= 2 {
343                    let addr = stack.pop().unwrap();
344                    let val = stack.pop().unwrap();
345                    let addr16 = format!("((_ extract 15 0) {})", addr);
346                    
347                    let new_present = self.fresh_var("present");
348                    writeln!(self.output, "(declare-const {} (Array (_ BitVec 16) (_ BitVec 64)))", new_present).unwrap();
349                    writeln!(self.output, "(assert (= {} (store {} {} {})))", new_present, present, addr16, val).unwrap();
350                    *present = new_present;
351                }
352            }
353            
354            OpCode::PresentRead => {
355                if let Some(addr) = stack.pop() {
356                    let addr16 = format!("((_ extract 15 0) {})", addr);
357                    stack.push(format!("(select {} {})", present, addr16));
358                }
359            }
360            
361            OpCode::Input => {
362                // Fresh symbolic input
363                let input_var = self.fresh_var("input");
364                writeln!(self.output, "(declare-const {} (_ BitVec 64))", input_var).unwrap();
365                stack.push(input_var);
366            }
367            
368            OpCode::Output => {
369                stack.pop();
370            }
371            
372            // Array opcodes - not fully supported in SMT encoding
373            OpCode::Pack | OpCode::Unpack | OpCode::Index | OpCode::Store => {
374                // Array operations require complex memory theory
375                // For now, we skip them in SMT encoding
376            }
377        }
378    }

380    /// Encode IF statement using ITE.
381    fn encode_if(&mut self, 
382                 then_branch: &[Stmt],
383                 else_branch: Option<&[Stmt]>,
384                 stack: &mut Vec<String>,
385                 present: &mut String,
386                 depth: usize) {
387        // Pop condition
388        let cond = stack.pop().unwrap_or("(_ bv0 64)".to_string());
389        let cond_bool = format!("(not (= {} (_ bv0 64)))", cond);
390        
391        // Save state
392        let stack_before = stack.clone();
393        let present_before = present.clone();
394        
395        // Encode then branch
396        self.encode_block(then_branch, stack, present, depth + 1);
397        let stack_then = stack.clone();
398        let present_then = present.clone();
399        
400        // Encode else branch
401        *stack = stack_before;
402        *present = present_before.clone();
403        
404        if let Some(else_stmts) = else_branch {
405            self.encode_block(else_stmts, stack, present, depth + 1);
406        }
407        let stack_else = stack.clone();
408        let present_else = present.clone();
409        
410        // Merge present using ITE
411        let merged_present = self.fresh_var("present");
412        writeln!(self.output, "(declare-const {} (Array (_ BitVec 16) (_ BitVec 64)))", merged_present).unwrap();
413        writeln!(self.output, "(assert (= {} (ite {} {} {})))", 
414                 merged_present, cond_bool, present_then, present_else).unwrap();
415        *present = merged_present;
416        
417        // Merge stacks (take minimum length, merge with ITE)
418        let min_len = stack_then.len().min(stack_else.len());
419        stack.clear();
420        for i in 0..min_len {
421            let merged_val = format!("(ite {} {} {})", cond_bool, stack_then[i], stack_else[i]);
422            stack.push(merged_val);
423        }
424    }

426    /// Encode WHILE statement via bounded unrolling.
427    fn encode_while(&mut self,
428                    cond: &[Stmt],
429                    body: &[Stmt],
430                    stack: &mut Vec<String>,
431                    present: &mut String,
432                    depth: usize) {
433        if depth >= self.max_unroll {
434            writeln!(self.output, "; Loop unroll limit reached").unwrap();
435            return;
436        }
437        
438        // Unroll: for i in 0..max_unroll: if cond then body
439        for _ in 0..self.max_unroll {
440            // Evaluate condition
441            self.encode_block(cond, stack, present, depth + 1);
442            
443            if stack.is_empty() {
444                break;
445            }
446            
447            // Encode as conditional body
448            let cond_val = stack.pop().unwrap();
449            let cond_bool = format!("(not (= {} (_ bv0 64)))", cond_val);
450            
451            // Save state
452            let stack_before = stack.clone();
453            let present_before = present.clone();
454            
455            // Encode body
456            self.encode_block(body, stack, present, depth + 1);
457            
458            // Merge: if cond was true, use new state; else keep old
459            let merged_present = self.fresh_var("present");
460            writeln!(self.output, "(declare-const {} (Array (_ BitVec 16) (_ BitVec 64)))", merged_present).unwrap();
461            writeln!(self.output, "(assert (= {} (ite {} {} {})))", 
462                     merged_present, cond_bool, present, present_before).unwrap();
463            *present = merged_present;
464            
465            // Merge stack
466            let min_len = stack.len().min(stack_before.len());
467            let new_stack: Vec<String> = (0..min_len)
468                .map(|i| format!("(ite {} {} {})", cond_bool, stack[i], stack_before[i]))
469                .collect();
470            *stack = new_stack;
471        }
472    }
