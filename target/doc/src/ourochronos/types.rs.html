1//! Compile-time type system for OUROCHRONOS.
2//!
3//! This module provides static analysis to track **temporal tainting** - which values
4//! depend on oracle reads (temporal) vs. which are pure constants (ground truth).
5//!
6//! # Type Lattice
7//!
8//! ```text
9//!        Temporal (depends on oracle)
10//!           ↑
11//!        Unknown (not yet determined)
12//!           ↑
13//!         Pure (constant, no oracle dependency)
14//! ```
15//!
16//! # Type Rules
17//!
18//! | Operation | Input Types | Output Type |
19//! |-----------|-------------|-------------|
20//! | `ORACLE` | `Pure` (address) | `Temporal` |
21//! | `PROPHECY` | `Any`, `Pure` (address) | - |
22//! | Arithmetic | `Temporal × Any` | `Temporal` |
23//! | Arithmetic | `Pure × Pure` | `Pure` |
24//! | Comparison | `Temporal × Any` | `Temporal` |
25//! | `IF` | `Temporal` condition | Both branches tainted |
26//!
27//! # Safety Properties
28//!
29//! The type checker enforces:
30//! 1. **Causal Integrity**: Temporal values must flow through verification
31//! 2. **Stability Guarantee**: Pure values cannot spontaneously become temporal
32//! 3. **Taint Tracking**: The programmer knows which computations depend on the future
33

38/// Temporal type for compile-time causal tracking.
39#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
40pub enum TemporalType {
41    /// Value has no temporal dependency (constant, ground truth).
42    Pure,
43    /// Value depends on at least one oracle read.
44    Temporal,
45    /// Type is not yet determined (for inference).
46    Unknown,
47}
48
49impl TemporalType {
50    /// Join two types in the lattice.
51    /// Temporal ⊔ anything = Temporal
52    /// Pure ⊔ Pure = Pure
53    /// Unknown ⊔ x = x
54    pub fn join(self, other: Self) -> Self {
55        match (self, other) {
56            (TemporalType::Temporal, _) | (_, TemporalType::Temporal) => TemporalType::Temporal,
57            (TemporalType::Pure, TemporalType::Pure) => TemporalType::Pure,
58            (TemporalType::Unknown, x) | (x, TemporalType::Unknown) => x,
59        }
60    }
61    
62    /// Check if this type is more specific than another.
63    pub fn is_subtype_of(self, other: Self) -> bool {
64        match (self, other) {
65            (_, TemporalType::Unknown) => true,
66            (TemporalType::Pure, TemporalType::Pure) => true,
67            (TemporalType::Temporal, TemporalType::Temporal) => true,
68            (TemporalType::Pure, TemporalType::Temporal) => true, // Pure can be used where Temporal expected
69            _ => false,
70        }
71    }
72    
73    /// Human-readable name.
74    pub fn name(&self) -> &'static str {
75        match self {
76            TemporalType::Pure => "Pure",
77            TemporalType::Temporal => "Temporal",
78            TemporalType::Unknown => "Unknown",
79        }
80    }
81}
82
83impl Default for TemporalType {
84    fn default() -> Self {
85        TemporalType::Unknown
86    }
87}

89/// Type error during checking.
90#[derive(Debug, Clone)]
91pub struct TypeError {
92    /// Description of the error.
93    pub message: String,
94    /// Location hint (statement index).
95    pub location: Option<usize>,
96}
97
98impl TypeError {
99    pub fn new(message: impl Into<String>) -> Self {
100        Self {
101            message: message.into(),
102            location: None,
103        }
104    }
105    
106    pub fn with_location(mut self, loc: usize) -> Self {
107        self.location = Some(loc);
108        self
109    }
110}
111
112impl std::fmt::Display for TypeError {
113    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
114        if let Some(loc) = self.location {
115            write!(f, "[stmt {}] {}", loc, self.message)
116        } else {
117            write!(f, "{}", self.message)
118        }
119    }
120}

122/// Result of type checking.
123#[derive(Debug, Clone)]
124pub struct TypeCheckResult {
125    /// Whether the program is well-typed.
126    pub is_valid: bool,
127    /// Any type errors found.
128    pub errors: Vec<TypeError>,
129    /// Warnings (non-fatal issues).
130    pub warnings: Vec<String>,
131    /// Inferred types for memory cells.
132    pub cell_types: HashMap<Address, TemporalType>,
133    /// Stack type at end of program.
134    pub final_stack_types: Vec<TemporalType>,
135}
136
137impl TypeCheckResult {
138    pub fn success() -> Self {
139        Self {
140            is_valid: true,
141            errors: Vec::new(),
142            warnings: Vec::new(),
143            cell_types: HashMap::new(),
144            final_stack_types: Vec::new(),
145        }
146    }
147    
148    pub fn with_error(error: TypeError) -> Self {
149        Self {
150            is_valid: false,
151            errors: vec![error],
152            warnings: Vec::new(),
153            cell_types: HashMap::new(),
154            final_stack_types: Vec::new(),
155        }
156    }
157}

159/// Type checker for OUROCHRONOS programs.
160pub struct TypeChecker {
161    /// Abstract stack of types.
162    stack: Vec<TemporalType>,
163    /// Types of memory cells (from PROPHECY writes).
164    cell_types: HashMap<Address, TemporalType>,
165    /// Collected errors.
166    errors: Vec<TypeError>,
167    /// Collected warnings.
168    warnings: Vec<String>,
169    /// Current statement index.
170    stmt_index: usize,
171    /// Whether we're in a temporal-controlled branch.
172    in_temporal_branch: bool,
173}
174
175impl TypeChecker {
176    /// Create a new type checker.
177    pub fn new() -> Self {
178        Self {
179            stack: Vec::new(),
180            cell_types: HashMap::new(),
181            errors: Vec::new(),
182            warnings: Vec::new(),
183            stmt_index: 0,
184            in_temporal_branch: false,
185        }
186    }
187    
188    /// Type-check a program.
189    pub fn check(&mut self, program: &Program) -> TypeCheckResult {
190        self.check_statements(&program.body);
191        
192        TypeCheckResult {
193            is_valid: self.errors.is_empty(),
194            errors: self.errors.clone(),
195            warnings: self.warnings.clone(),
196            cell_types: self.cell_types.clone(),
197            final_stack_types: self.stack.clone(),
198        }
199    }
200    
201    /// Check a block of statements.
202    fn check_statements(&mut self, stmts: &[Stmt]) {
203        for stmt in stmts {
204            self.check_stmt(stmt);
205            self.stmt_index += 1;
206        }
207    }
208    
209    /// Check a single statement.
210    fn check_stmt(&mut self, stmt: &Stmt) {
211        match stmt {
212            Stmt::Push(value) => {
213                // Constants are always pure
214                let ty = if value.prov.is_pure() {
215                    TemporalType::Pure
216                } else {
217                    TemporalType::Temporal
218                };
219                self.stack.push(ty);
220            }
221            
222            Stmt::Op(op) => {
223                self.check_op(*op);
224            }
225            
226            Stmt::If { then_branch, else_branch } => {
227                // Pop condition
228                let cond_type = self.pop_type();
229                
230                // If condition is temporal, both branches are tainted
231                let was_temporal = self.in_temporal_branch;
232                if cond_type == TemporalType::Temporal {
233                    self.in_temporal_branch = true;
234                }
235                
236                // Check branches
237                let stack_before = self.stack.clone();
238                self.check_statements(then_branch);
239                let then_stack = self.stack.clone();
240                
241                self.stack = stack_before;
242                if let Some(else_stmts) = else_branch {
243                    self.check_statements(else_stmts);
244                }
245                let else_stack = self.stack.clone();
246                
247                // Merge branch stacks (join types)
248                self.stack = self.merge_stacks(&then_stack, &else_stack);
249                
250                self.in_temporal_branch = was_temporal;
251            }
252            
253            Stmt::While { cond, body } => {
254                // Check condition
255                let stack_before = self.stack.clone();
256                self.check_statements(cond);
257                let cond_type = self.pop_type();
258                
259                // Loops with temporal conditions are especially tricky
260                if cond_type == TemporalType::Temporal {
261                    self.warnings.push(format!(
262                        "Loop condition is Temporal at stmt {}: loop behavior may depend on future",
263                        self.stmt_index
264                    ));
265                    self.in_temporal_branch = true;
266                }
267                
268                // Check body
269                self.check_statements(body);
270                
271                // Restore stack (approximate: assume loop terminates)
272                self.stack = stack_before;
273            }
274            
275            Stmt::Block(stmts) => {
276                self.check_statements(stmts);
277            }
278            
279            Stmt::Call { name: _ } => {
280                // Procedure call: we don't have the body here, so mark as Unknown
281                // In a full implementation, we'd look up the procedure and analyze its body
282                self.stack.push(TemporalType::Unknown);
283            }
284            
285            Stmt::Match { cases, default } => {
286                // Pop the matched value
287                self.pop_type();
288                
289                // Analyze all branches (conservative: union of all possible types)
290                for (_, body) in cases {
291                    for stmt in body {
292                        self.check_stmt(stmt);
293                    }
294                }
295                if let Some(default_body) = default {
296                    for stmt in default_body {
297                        self.check_stmt(stmt);
298                    }
299                }
300            }
301            
302            Stmt::TemporalScope { body, .. } => {
303                // Temporal scope: check the body
304                // Memory isolation doesn't affect type checking at this level
305                self.check_statements(body);
306            }
307        }
308    }
309    
310    /// Check an opcode.
311    fn check_op(&mut self, op: OpCode) {
312        match op {
313            // ===== Temporal Operations =====
314            OpCode::Oracle => {
315                let addr_type = self.pop_type();
316                if addr_type == TemporalType::Temporal {
317                    self.warnings.push(format!(
318                        "ORACLE address is Temporal at stmt {}: address depends on future",
319                        self.stmt_index
320                    ));
321                }
322                // Result is always temporal
323                self.stack.push(TemporalType::Temporal);
324            }
325            
326            OpCode::Prophecy => {
327                let addr_type = self.pop_type();
328                let value_type = self.pop_type();
329                
330                if addr_type == TemporalType::Temporal {
331                    self.warnings.push(format!(
332                        "PROPHECY address is Temporal at stmt {}: writing to future-dependent address",
333                        self.stmt_index
334                    ));
335                }
336                
337                // Record the type written to this cell (if address is constant)
338                // In practice, we don't know the address statically, so we approximate
339                self.cell_types.insert(0, value_type); // Simplified: assume address 0
340            }
341            
342            OpCode::PresentRead => {
343                let addr_type = self.pop_type();
344                // Reading from present can be temporal if written with temporal value
345                let result = if self.in_temporal_branch {
346                    TemporalType::Temporal
347                } else {
348                    addr_type // Approximate: same as address type
349                };
350                self.stack.push(result);
351            }
352            
353            // ===== Stack Manipulation =====
354            OpCode::Pop => {
355                self.pop_type();
356            }
357            
358            OpCode::Dup => {
359                let ty = self.peek_type();
360                self.stack.push(ty);
361            }
362            
363            OpCode::Swap => {
364                if self.stack.len() >= 2 {
365                    let len = self.stack.len();
366                    self.stack.swap(len - 1, len - 2);
367                }
368            }
369            
370            OpCode::Over => {
371                if self.stack.len() >= 2 {
372                    let ty = self.stack[self.stack.len() - 2];
373                    self.stack.push(ty);
374                }
375            }
376            
377            OpCode::Rot => {
378                if self.stack.len() >= 3 {
379                    let len = self.stack.len();
380                    let a = self.stack[len - 3];
381                    self.stack[len - 3] = self.stack[len - 2];
382                    self.stack[len - 2] = self.stack[len - 1];
383                    self.stack[len - 1] = a;
384                }
385            }
386            
387            OpCode::Depth => {
388                // Depth is always pure (it's a constant at that point)
389                self.stack.push(TemporalType::Pure);
390            }
391            
392            OpCode::Pick => {
393                let _ = self.pop_type(); // index
394                // Result depends on what we're picking
395                // Approximate: could be temporal
396                self.stack.push(TemporalType::Unknown);
397            }
398            
399            OpCode::Nop | OpCode::Neg => {
400                // Nop: no effect. Neg: unary, preserves type.
401            }
402            
403            // ===== Arithmetic (joins types) =====
404            OpCode::Add | OpCode::Sub | OpCode::Mul | OpCode::Div | OpCode::Mod => {
405                let b = self.pop_type();
406                let a = self.pop_type();
407                self.stack.push(a.join(b));
408            }
409            
410            // ===== Bitwise (joins types) =====
411            OpCode::And | OpCode::Or | OpCode::Xor | OpCode::Shl | OpCode::Shr => {
412                let b = self.pop_type();
413                let a = self.pop_type();
414                self.stack.push(a.join(b));
415            }
416            
417            OpCode::Not => {
418                // Unary: preserves type
419                // Type stays the same
420            }
421            
422            // ===== Comparison (joins types) =====
423            OpCode::Eq | OpCode::Neq | OpCode::Lt | OpCode::Gt | OpCode::Lte | OpCode::Gte => {
424                let b = self.pop_type();
425                let a = self.pop_type();
426                self.stack.push(a.join(b));
427            }
428            
429            // ===== I/O =====
430            OpCode::Input => {
431                // Input is pure (external, not from oracle)
432                self.stack.push(TemporalType::Pure);
433            }
434            
435            OpCode::Output => {
436                let ty = self.pop_type();
437                if ty == TemporalType::Temporal && !self.in_temporal_branch {
438                    // Outputting temporal value: this is significant
439                    // The output depends on the fixed point
440                }
441            }
442            
443            // ===== Control =====
444            OpCode::Halt | OpCode::Paradox => {
445                // No stack effect
446            }
447            
448            // ===== Arrays =====
449            OpCode::Pack => {
450                // Consumes n+2 values, produces none
451                self.pop_type(); // n
452                self.pop_type(); // base
453                // Additional values popped dynamically
454            }
455            OpCode::Unpack => {
456                // Consumes 2, produces n values - type is unknown
457                self.pop_type(); // n
458                self.pop_type(); // base
459                self.stack.push(TemporalType::Unknown);
460            }
461            OpCode::Index => {
462                // Read from memory - result may be temporal
463                let _index_type = self.pop_type();
464                let _base_type = self.pop_type();
465                self.stack.push(TemporalType::Temporal);
466            }
467            OpCode::Store => {
468                // Write to memory
469                self.pop_type(); // index
470                self.pop_type(); // base
471                self.pop_type(); // value
472            }
473        }
474    }
475    
476    /// Pop a type from the abstract stack.
477    fn pop_type(&mut self) -> TemporalType {
478        self.stack.pop().unwrap_or(TemporalType::Unknown)
479    }
480    
481    /// Peek at the top type.
482    fn peek_type(&self) -> TemporalType {
483        self.stack.last().copied().unwrap_or(TemporalType::Unknown)
484    }
485    
486    /// Merge two stacks from branches (join corresponding types).
487    fn merge_stacks(&self, a: &[TemporalType], b: &[TemporalType]) -> Vec<TemporalType> {
488        let max_len = a.len().max(b.len());
489        let mut result = Vec::with_capacity(max_len);
490        
491        for i in 0..max_len {
492            let ta = a.get(i).copied().unwrap_or(TemporalType::Unknown);
493            let tb = b.get(i).copied().unwrap_or(TemporalType::Unknown);
494            result.push(ta.join(tb));
495        }
496        
497        result
498    }

507/// Type-check a program and return the result.
508pub fn type_check(program: &Program) -> TypeCheckResult {
509    let mut checker = TypeChecker::new();
510    checker.check(program)
511}

513/// Display type information for a program.
514pub fn display_types(result: &TypeCheckResult) -> String {
515    let mut output = String::new();
516    
517    output.push_str("=== Type Check Result ===\n");
518    output.push_str(&format!("Valid: {}\n", result.is_valid));
519    
520    if !result.errors.is_empty() {
521        output.push_str("\nErrors:\n");
522        for err in &result.errors {
523            output.push_str(&format!("  - {}\n", err));
524        }
525    }
526    
527    if !result.warnings.is_empty() {
528        output.push_str("\nWarnings:\n");
529        for warn in &result.warnings {
530            output.push_str(&format!("  - {}\n", warn));
531        }
532    }
533    
534    if !result.cell_types.is_empty() {
535        output.push_str("\nCell Types:\n");
536        for (addr, ty) in &result.cell_types {
537            output.push_str(&format!("  [{}]: {}\n", addr, ty.name()));
538        }
539    }
540    
541    if !result.final_stack_types.is_empty() {
542        output.push_str("\nFinal Stack Types:\n");
543        for (i, ty) in result.final_stack_types.iter().enumerate() {
544            output.push_str(&format!("  {}: {}\n", i, ty.name()));
545        }
546    }
547    
548    output
549}

594        assert_eq!(TemporalType::Temporal.join(TemporalType::Pure), TemporalType::Temporal);
